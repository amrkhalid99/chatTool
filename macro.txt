Sub ExportCleanJson()
    Dim lastRow As Long
    Dim i As Long
    Dim jsonText As String
    Dim ws As Worksheet
    
    Set ws = ActiveSheet
    lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row
    
    ' بداية JSON
    jsonText = "[" & vbCrLf
    
    For i = 2 To lastRow
        ' السطر ده مهم: بننضف أي حرف ممكن يخرب JSON
        Dim idVal As String
        Dim titleVal As String
        Dim catVal As String
        Dim contVal As String
        Dim pinnedVal As String
        
        ' بنشيل أي حرف double quote جوه النص عشان يبوظ JSON
        idVal = Replace(ws.Cells(i, 1).Value, """", "\""")
        titleVal = Replace(ws.Cells(i, 2).Value, """", "\""")
        catVal = Replace(ws.Cells(i, 3).Value, """", "\""")
        contVal = Replace(ws.Cells(i, 4).Value, """", "\""")
        
        ' الخانة الخامسة (pinned) - بنحولها لقيمة boolean
        If i <= lastRow And ws.Cells(i, 5).Value <> "" Then
            ' لو مكتوب TRUE أو YES أو 1 أو أي حاجة إيجابية
            Dim pinValue As String
            pinValue = LCase(Trim(ws.Cells(i, 5).Value))
            If pinValue = "true" Or pinValue = "yes" Or pinValue = "1" Or pinValue = "صح" Or pinValue = "نعم" Then
                pinnedVal = "true"
            Else
                pinnedVal = "false"
            End If
        Else
            pinnedVal = "false"  ' افتراضي false لو الخلية فاضية
        End If
        
        ' بنشيل الـ Enter من جوه النص عشان ميقطعش JSON
        contVal = Replace(contVal, vbCrLf, " ")
        contVal = Replace(contVal, vbLf, " ")
        contVal = Replace(contVal, vbCr, " ")
        
        ' تعديل الترتيب: content قبل category
        jsonText = jsonText & "  {"
        jsonText = jsonText & """id"": """ & idVal & """, "
        jsonText = jsonText & """title"": """ & titleVal & """, "
        jsonText = jsonText & """content"": """ & contVal & """, "
        jsonText = jsonText & """category"": """ & catVal & """, "
        jsonText = jsonText & """pinned"": " & pinnedVal & ""  ' هنا بدون علامات اقتباس لأنها boolean
        jsonText = jsonText & "}"
        
        ' لو مش آخر سطر، ضيف فاصلة، مفيش فاصلة بعد آخر سطر
        If i < lastRow Then
            jsonText = jsonText & "," & vbCrLf
        Else
            jsonText = jsonText & vbCrLf
        End If
    Next i
    
    jsonText = jsonText & "]"
    
    ' نسخ للحافظة بطريقة مضمونة
    Dim obj As Object
    Set obj = CreateObject("htmlfile")
    obj.ParentWindow.ClipboardData.SetData "text", jsonText
    
    ' رسالة تأكيد
    MsgBox "JSON نضيف 100%!" & vbNewLine & "افتح JSON Formatter أو Notepad واعمل Paste", vbInformation
    
    ' للتحقق: نشوف أول 100 حرف
    Debug.Print Left(jsonText, 200)
End Sub
